// FOR LOOP
#include <iostream>
using namespace std;
int main(){
    for(int i=0;i<=100;i++){
       cout<<i<<endl;
    }
    return 0;
}
// EXAMPLE OF INFINITE LOOP
#include <iostream>
using namespace std;
int main(){
    for (int i=0;34<=40;i++)
    {
        cout<<i<<endl;
    }
    return 0;
}
// WHILE LOOP
#include <iostream>
using namespace std;
int main(){
    int i=1;
    while(i<=40){
        cout<<i<<endl;
        i++;
    }
    return 0;
}
// EXAMPLE OF INFINITE WHILE LOOP
#include <iostream>
using namespace std;
int main(){
    int i=1;
    while(true){
        cout<<i<<endl;
        
    }
    return 0;
}
// DO-WHILE LOOP
#include <iostream>
using namespace std;
int main(){
    int i=1;
    do
    {
       cout<<i<<endl;
       i++;
    }
     while (i>=40);
    
    return 0;

}
